1  /*
2  * @brief peclet solves the convection-diffusion equation
3  *
4  * @detail
5  *
6  * This solves the unsteady convection-diffusion equation.
7  *
8  * System assembly and time stepping are based on deal.II Tutorial 26 by Wolfgang Bangerth, Texas A&M University, 2013
9  *
10  * Some of the more notable extensions include:
11  * - Builds convection-diffusion matrix instead of Laplace matrix
12  * - Supports time-dependent non-zero Dirichlet and Neumann boundary condition
13  * - Re-designed parmameter handling
14  * - Generalized boundary condition handling via the parameter input file
15  * - Writes FEFieldFunction to disk, and can read it from disk to initialize a restart
16  * - Extended the FEFieldFunction class for extrapolation
17  * - Added verification via Method of Manufactured Solutions (MMS) with error table based on approach from Tutorial 7
18  * - Added test suite using ctest and the standard deal.II approach
19  * - Added a parameteric sphere-cylinder grid
20  * - Added a boundary grid refinement routine
21  * - Added a output option for 1D solutions in tabular format
22  *
23  * @author Alexander Zimmerman <zimmerman@aices.rwth-aachen.de>, RWTH AAchen University, 2016
24  */

210  SolverStatus Peclet<dim>::solve_time_step(bool quiet)
211  {
212  double tolerance = this->params.solver.tolerance;
213  if (this->params.solver.normalize_tolerance)
214  {
215  tolerance *= this->system_rhs.l2_norm();
216  }
217  SolverControl solver_control(
218  this->params.solver.max_iterations,
219  tolerance);
220 
221  SolverCG<> solver_cg(solver_control);
222  SolverBicgstab<> solver_bicgstab(solver_control);
223 
224  PreconditionSSOR<> preconditioner;
225 
226  preconditioner.initialize(this->system_matrix, 1.0);
227 
228  std::string solver_name;
229 
230  if (this->params.solver.method == "CG")
231  {
232  solver_name = "CG";
233  solver_cg.solve(
234  this->system_matrix,
235  this->solution,
236  this->system_rhs,
237  preconditioner);
238  }
239  else if (this->params.solver.method == "BiCGStab")
240  {
241  solver_name = "BiCGStab";
242  solver_bicgstab.solve(
243  this->system_matrix,
244  this->solution,
245  this->system_rhs,
246  preconditioner);
247  }
248 
249  this->constraints.distribute(this->solution);
250 
251  if (!quiet)
252  {
253  std::cout << " " << solver_control.last_step()
254  << " " << solver_name << " iterations." << std::endl;
255  }
256 
257  SolverStatus status;
258  status.last_step = solver_control.last_step();
259 
260  return status;
261

368  /*
369  Working with deal.II's Function class has been interesting, and I'm
370  sure many of my choices are unorthodox. The most important lesson learned has been that
371  a Function<dim>* can point to any class derived from Function<dim>. The general design pattern
372  then is to instantitate all of the functions that might be needed, and then to point to the ones
373  actually being used.
374  */
375  Functions::ParsedFunction<dim> parsed_velocity_function(dim),
376  parsed_diffusivity_function,
377  parsed_source_function,
378  parsed_boundary_function,
379  parsed_initial_values_function,
380  parsed_exact_solution_function;
381 
382  this->params = Parameters::read<dim>(
383  parameter_file,
384  parsed_velocity_function,
385  parsed_diffusivity_function,
386  parsed_source_function,
387  parsed_boundary_function,
388  parsed_exact_solution_function,
389  parsed_initial_values_function);
390 
391  this->create_coarse_grid();
392 
393  this->velocity_function = &parsed_velocity_function;
394  this->diffusivity_function = &parsed_diffusivity_function;
395  this->source_function = &parsed_source_function;
396  this->exact_solution_function = &parsed_exact_solution_function;
397 
398  /*
399  Generalizing the handling of auxiliary functions is complicated. In most cases one should be able to use a ParsedFunction, but the generality of Function<dim>* allows for a standard way to account for any possible derived class of Function<dim>.
400  For example this allows for....
401  - an optional initial values function that interpolates an old solution loaded from disk.
402  - flexibily implementing general boundary conditions
403  */
404 
405  // Initial values function
406 
407  Triangulation<dim> field_grid;
408  DoFHandler<dim> field_dof_handler(field_grid);
409  Vector<double> field_solution;
410 
411  if (this->params.initial_values.function_name != "interpolate_old_field")
412  { // This will write files that need to exist.
413  this->setup_system(true);
414  FEFieldTools::save_field_parts(this->triangulation, this->dof_handler, this->solution);
415  }
416 
417  FEFieldTools::load_field_parts(
418  field_grid,
419  field_dof_handler,
420  field_solution,
421  this->fe);
422 
423  MyFunctions::ExtrapolatedField<dim> field_function(
424  field_dof_handler,
425  field_solution);
426 
427 
428  if (this->params.initial_values.function_name == "interpolate_old_field")
429  {
430  this->initial_values_function = &field_function;
431  }
432  else if (this->params.initial_values.function_name == "parsed")
433  {
434  this->initial_values_function = &parsed_initial_values_function;
435  }
436 
437  // Boundary condition functions
438 
439  unsigned int boundary_count = this->params.boundary_conditions.implementation_types.size();
440 
441  assert(params.boundary_conditions.function_names.size() == boundary_count);
442 
443  std::vector<ConstantFunction<dim>> constant_functions;
444 
445  for (unsigned int boundary = 0; boundary < boundary_count; boundary++)
446  {
447  std::string boundary_type = this->params.boundary_conditions.implementation_types[boundary];
448  std::string function_name = this->params.boundary_conditions.function_names[boundary];
449 
450  if (function_name == "constant")
451  {
452  double value = this->params.boundary_conditions.function_double_arguments.front();
453  this->params.boundary_conditions.function_double_arguments.pop_front();
454  constant_functions.push_back(ConstantFunction<dim>(value));
455  }
456  }
457 
458  // Organize boundary functions to simplify application during the time loop
459 
460  unsigned int constant_function_index = 0;
461 
462  for (unsigned int boundary = 0; boundary < boundary_count; boundary++)
463  {
464  std::string boundary_type = this->params.boundary_conditions.implementation_types[boundary];
465  std::string function_name = this->params.boundary_conditions.function_names[boundary];
466 
467  if (function_name == "constant")
468  {
469  assert(constant_function_index < constant_functions.size());
470  this->boundary_functions.push_back(&constant_functions[constant_function_index]);
471  constant_function_index++;
472  }
473  else if (function_name == "parsed")
474  {
475  this->boundary_functions.push_back(&parsed_boundary_function);
476  }
477 
478  }
479 
480  // Attach manifolds
481 
482  assert(dim < 3); // @todo: 3D extension: For now the CylindricalManifold is being ommitted.
483  // deal.II makes is impractical for a CylindricalManifold to exist in 2D.
484  SphericalManifold<dim> spherical_manifold(this->spherical_manifold_center);
485 
486  for (unsigned int i = 0; i < manifold_ids.size(); i++)
487  {
488  if (manifold_descriptors[i] == "spherical")
489  {
490  this->triangulation.set_manifold(manifold_ids[i], spherical_manifold);
491  }
492  }
493 
494  // Run initial refinement cycles
495 
496  this->triangulation.refine_global(this->params.refinement.initial_global_cycles);
497 
498  Refinement::refine_mesh_near_boundaries(
499  this->triangulation,
500  this->params.refinement.boundaries_to_refine,
501  this->params.refinement.initial_boundary_cycles);
502 
503  // Initialize the linear system
504 
505  this->setup_system();
506 
507  Vector<double> tmp;
508  Vector<double> forcing_terms;
509 
510  // Iterate
511  unsigned int pre_refinement_step = 0;
512 
513 start_time_iteration:
514 
515  tmp.reinit(this->solution.size());
516 
517  VectorTools::interpolate(this->dof_handler,
518  *this->initial_values_function,
519  this->old_solution);
520 
521  this->solution = this->old_solution;
522  this->time_step_counter = 0;
523  this->time = 0;
524 
525  double theta = this->params.time.semi_implicit_theta;
526  this->time_step_size = this->params.time.step_size;
527  if (this->time_step_size < EPSILON)
528  {
529  this->time_step_size = this->params.time.end_time/
530  pow(2., this->params.time.global_refinement_levels);
531  }
532  double Delta_t = this->time_step_size;
533 
534  this->write_solution();
535  bool final_time_step = false;
536  bool output_this_step = true;
537  double epsilon = 1e-14;
538 
539  SolverStatus solver_status;
540 
541  do
542  {
543  ++this->time_step_counter;
544  time = Delta_t*time_step_counter; // Incrementing the time directly would accumulate errors
545 
546  // Set some flags that will control output for this step.
547  final_time_step = this->time > this->params.time.end_time - epsilon;
548 
549  bool at_interval = false;
550 
551  if (this->params.output.time_step_interval == 1)
552  {
553  at_interval = true;
554  }
555  else if (this->params.output.time_step_interval != 0)
556  {
557  if ((time_step_counter % this->params.output.time_step_interval) == 0)
558  {
559  at_interval = true;
560  }
561  }
562  else
563  {
564  at_interval = false;
565  }
566 
567  if (at_interval)
568  {
569  output_this_step = true;
570  }
571  else
572  {
573  output_this_step = false;
574  }
575 
576  // Run the time step
577  if (output_this_step)
578  {
579  std::cout << "Time step " << this->time_step_counter
580  << " at t=" << this->time << std::endl;
581  }
582 
583  // Add mass and convection-diffusion matrix terms to RHS
584  this->mass_matrix.vmult(this->system_rhs, this->old_solution);
585 
586  this->convection_diffusion_matrix.vmult(tmp, this->old_solution);
587 
588  this->system_rhs.add(-(1. - theta) * Delta_t, tmp);
589 
590  // Add source terms to RHS
591  source_function->set_time(this->time);
592  VectorTools::create_right_hand_side(this->dof_handler,
593  QGauss<dim>(fe.degree+1),
594  *source_function,
595  tmp);
596  forcing_terms = tmp;
597  forcing_terms *= Delta_t * theta;
598 
599  source_function->set_time(this->time - Delta_t);
600  VectorTools::create_right_hand_side(this->dof_handler,
601  QGauss<dim>(fe.degree+1),
602  *source_function,
603  tmp);
604  forcing_terms.add(Delta_t * (1 - theta), tmp);
605 
606  this->system_rhs += forcing_terms;
607 
608  // Add natural boundary conditions to RHS
609  for (unsigned int boundary = 0; boundary < boundary_count; boundary++)
610  {
611  if ((this->params.boundary_conditions.implementation_types[boundary] != "natural"))
612  {
613  continue;
614  }
615 
616  std::set<types::boundary_id> dealii_boundary_id = {boundary}; // @todo: This throws a warning
617 
618  boundary_functions[boundary]->set_time(this->time);
619 
620  MyVectorTools::my_create_boundary_right_hand_side(
621  this->dof_handler,
622  QGauss<dim-1>(fe.degree+1),
623  *boundary_functions[boundary],
624  tmp,
625  dealii_boundary_id);
626  forcing_terms = tmp;
627  forcing_terms *= Delta_t * theta;
628 
629  boundary_functions[boundary]->set_time(this->time - Delta_t);
630  MyVectorTools::my_create_boundary_right_hand_side(
631  this->dof_handler,
632  QGauss<dim-1>(fe.degree+1),
633  *boundary_functions[boundary],
634  tmp,
635  dealii_boundary_id);
636 
637  forcing_terms.add(Delta_t * (1. - theta), tmp);
638 
639  this->system_rhs += forcing_terms;
640 
641  }
642 
643  // Make the system matrix and apply constraints
644  system_matrix.copy_from(mass_matrix);
645 
646  system_matrix.add(theta * Delta_t, convection_diffusion_matrix);
647 
648  constraints.condense (system_matrix, system_rhs);
649 
650  {
651  // Apply strong boundary conditions
652  std::map<types::global_dof_index, double> boundary_values;
653  for (unsigned int boundary = 0; boundary < boundary_count; boundary++)
654  {
655  if (this->params.boundary_conditions.implementation_types[boundary] != "strong")
656  {
657  continue;
658  }
659 
660  boundary_functions[boundary]->set_time(time);
661 
662  VectorTools::interpolate_boundary_values
663  (
664  this->dof_handler,
665  boundary,
666  *boundary_functions[boundary],
667  boundary_values
668  );
669  }
670  MatrixTools::apply_boundary_values(
671  boundary_values,
672  this->system_matrix,
673  this->solution,
674  this->system_rhs);
675  }
676 
677  solver_status = this->solve_time_step(!output_this_step);
678 
679  if ((this->params.time.stop_when_steady) & (solver_status.last_step == 0))
680  {
681  std::cout << "Reached steady state at t = " << this->time << std::endl;
682  final_time_step = true;
683  output_this_step = true;
684  }
685 
686  if (output_this_step)
687  {
688  this->write_solution();
689 
690  if (this->params.verification.enabled)
691  {
692  this->append_verification_table();
693  }
694 
695  }
696 
697  if ((time_step_counter == 1) &&
698  (pre_refinement_step < this->params.refinement.adaptive.initial_cycles))
699  {
700  this->adaptive_refine();
