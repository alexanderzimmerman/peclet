97  /*
98 
99  @brief Create the convection-diffusion matrix
100 
101  @detail
102 
103  This is the sum of the convention and diffusions matrices, C + K,
104  for the convection-diffusion equation in
105  Finite Element Methods for Flow Problems (Donea & Huerta, 2003)
106 
107  K is equivalent to the Laplace matrix, and so most of this is
108  copied from dealii::MatrixCreator::create_laplace_matrix
109 
110  Super important note: The convection operator is asymmetric.
111 
112  @author A. Zimmerman <zimmerman@aices.rwth-aachen.de>
113 
114  */
115  template <int dim,
116  typename CellIterator>
117  void convection_diffusion_assembler (
118  const CellIterator &cell,
119  AssemblerData::Scratch<dim,double> &data,
120  MatrixCreator::internal::AssemblerData::CopyData<double> &copy_data)
121  {
122  data.x_fe_values.reinit (cell);
123  const FEValues<dim> &fe_values = data.x_fe_values.get_present_fe_values ();
124 
125  const unsigned int dofs_per_cell = fe_values.dofs_per_cell,
126  n_q_points = fe_values.n_quadrature_points;
127 
128  const FiniteElement<dim> &fe = fe_values.get_fe();
129 
130  Assert(data.diffusivity->n_components==1,
131  ::dealii::MatrixCreator::ExcComponentMismatch());
132 
133  Assert(data.convection_velocity->n_components==dim,
134  ::dealii::MatrixCreator::ExcComponentMismatch());
135 
136  copy_data.cell_matrix.reinit (dofs_per_cell, dofs_per_cell);
137 
138  copy_data.dof_indices.resize (dofs_per_cell);
139 
140  cell->get_dof_indices (copy_data.dof_indices);
141 
142 
143  data.diffusivity_values.resize (n_q_points);
144 
145  data.diffusivity->value_list (fe_values.get_quadrature_points(),
146  data.diffusivity_values);
147 
148  data.convection_velocity_values.resize (n_q_points,
149  dealii::Vector<double>(dim));
150 
151  data.convection_velocity->vector_value_list (fe_values.get_quadrature_points(),
152  data.convection_velocity_values);
153 
154 
155  const std::vector<double> &JxW = fe_values.get_JxW_values();
156 
157  double add_data;
158 
159  assert(fe.is_primitive());
160 
161  for (unsigned int i=0; i<dofs_per_cell; ++i)
162  {
163  const Tensor<1,dim> *grad_phi_i = &fe_values.shape_grad(i,0);
164  const double *phi_i = &fe_values.shape_value(i,0);
165 
166  for (unsigned int j=0; j<dofs_per_cell; ++j)
167  {
168  const Tensor<1,dim> *grad_phi_j = &fe_values.shape_grad(j,0);
169 
170  add_data = 0;
171 
172  for (unsigned int point=0; point<n_q_points; ++point)
173  {
174  add_data += (grad_phi_i[point]*grad_phi_j[point]) *
175  JxW[point] *
176  data.diffusivity_values[point];
177 
178  Tensor<1,dim> a;
179  for (unsigned int ia = 0; ia < dim; ia++)
180  { // @todo: What is the proper way to convert this?
181  a[ia] = data.convection_velocity_values[point][ia];
182  }
183 
184  add_data += (phi_i[point] * ( a * grad_phi_j[point]) ) * JxW[point];
185 
186  }
187 
188  copy_data.cell_matrix(i,j) = add_data;
189 
190  }
191 
192  }
